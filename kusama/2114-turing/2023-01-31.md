```bash
SELECT date(block_time) as logDT, MIN(number) startBN, MAX(number) endBN, COUNT(*) numBlocks 
 FROM `substrate-etl.kusama.blocks2114`  
 where LAST_DAY(date(block_time)) = "2023-01-31" 
 group by logDT 
 order by logDT
```
